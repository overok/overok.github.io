# for 循环 let 与 var 的区别？


var: 变量i是var声明的,变量里i会重复,在全局范围内都有效,所以全局只有一个变量i,因而最后打印会是重复值。
let: 变量i是let声明的,当前的i只在本轮循环有效,所以每一次循环的i其实都是一个新的变量,所以最后输出值不一样。
//如果每一轮循环的变量i都是重新声明的,那它怎么知道上一轮循环的值?
这是因为 JavaScript 引擎内部会记住上一轮循环的值，初始化本轮的变量i时，就在上一轮循环的基础上进行计算。

# call 、apply 和 bind

//call 、apply 和 bind 是挂在 Function 对象上的三个方法，调用这三个方法的必须是一个函数。
//bind 虽然改变了 func 的 this 指向，但不是马上执行，而call、apply是在改变了函数的 this 指向之后立马执行。
1.func.call(thisArg, param1, param2, ...)
2.func.apply(thisArg, [param1,param2,...])
3.func.bind(thisArg, param1, param2, ...)
//手写call
function myCall(ctx,...args) {
  ctx = ctx || window;
  ctx.fn = this;
  const res = ctx.fn(...args);
  delete ctx.fn;
  return res;
}
Function.prototype.myCall= myCall;
console.log.myCall(null,1,2);
//手写apply
function myCall(ctx,args) {
  ctx = ctx || window;
  ctx.fn = this;
  const res = ctx.fn(...args);
  delete ctx.fn;
  return res;
}
Function.prototype.myCall= myCall;
console.log.myCall(null,[1,2]);
//手写bind
function myCall(ctx,args) {
  const fn = this;
  return function Fn() {
    return fn.apply(
      this instanceof Fn ? this : ctx,
      // 当前的这个 arguments 是指 Fn 的参数
      args.concat(...arguments)
    );
  };
}

# 取出数组元素最多的一项

function findMost(arr) {
  if (!arr.length) return
  if (arr.length === 1) return 1
  let res = {}
  let maxName,
    maxNum = 0
  // 遍历数组
  arr.forEach((item) => {
    res[item] ? (res[item] += 1) : (res[item] = 1)
    if (res[item] > maxNum) {
      maxName = item
      maxNum = res[item]
    }
  })
  return '出现次数最多的元素为:' + maxName + ', 出现次数为:' + maxNum
}

# 计算一个对象的层数

const obj = { c: { d: { e: { f: 1 } } } } // 4
//手写函数
function loopGetLevel(obj) {
  var res = 1
  function computedLevel(obj, level) {
    var level = level ? level : 0
    if (typeof obj === 'object') {
      for (var key in obj) {
        if (typeof obj[key] === 'object') {
          computedLevel(obj[key], level + 1)
        } else {
          res = level + 1 > res ? level + 1 : res
        }
      }
    } else {
      res = level > res ? level : res
    }
  }
  computedLevel(obj)
  return res
}

# for...in 和 for...of 的区别?

  • for...of 遍历获取的是对象的键值, for...in 获取的是对象的键名;
  • for....of 循环可以用来遍历数组、类数组对象、字符串、Set、Map 以及 Generator 对象
  • for...in 会遍历对象的整个原型链, 性能非常差不推荐使用,而 for...of 只遍历当前对象不会遍历原型链;
  • for...in 循环主要是为了遍历对象而生,不适用遍历数组;

# 实现 add 函数

//add(1)(2)(3)()=6
function add(...args1) {
  let allArgs = [...args1]
  function fn(...args2) {
    if (!args2.length) return fn.toString()
    allArgs = [...allArgs, ...args2]
    return fn
  }
  fn.toString = function() {
    return allArgs.reduce((pre, next) => pre + next)
  }
  return fn
}

function unique1(array){

3.浅拷贝实现原理：
function shallowCopy(obj) {
  var target = {};
  for (let i in obj) {
    if (obj.hasOwnproperty(i)) {
      target[i] = obj[i];
    }
  }
  return target;
}
//深拷贝

2.通过递归来实现一个深拷贝
  function deepCopy(obj) {
    if (obj == null)  return obj;
    if (obj instanceof Date) return new Date();
    if (obj instanceof RegExp) return new RegExp(obj);
    if (typeof obj !== 'object') return obj;
    let cloneObj = new obj.constructor();
    for (let i in obj) {
      if (obj.hasOwnproperty(i)) {
        cloneObj[i] = deepCopy(obj[i]);
      }
    }
    return cloneObj;
  }
