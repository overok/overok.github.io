# 拼接字符串

//旧
const name = '小明'
const score = 59
let result = ''
if (score > 60) {
  result = `${name}的考试成绩及格`
} else {
  result = `${name}的考试成绩不及格`
}
//新
const name = '小明'
const score = 59
const result = `${name}${score > 60 ? '的考试成绩及格' : '的考试成绩不及格'}`
